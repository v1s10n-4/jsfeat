**Goal:** Build a dev-only debug page (`/#/dev`) that runs the exact same card detection pipeline on static test images and live webcam, with full-resolution pipeline overlays, parameter controls, batch testing, and regression tracking.

**Architecture:** New lazy-loaded route renders DevPage which orchestrates a DebugCanvas (runs `cardDetectionDemo.process()` with overlays), PipelineStages thumbnails, DetectionPanel (controls + metrics), and TestImageStrip (batch runner). A `getCardDebugBuffers()` export from `demos.ts` exposes internal pipeline state. Test image manifest generated from filesystem scan.

**Tech Stack:** React 19, React Router 7, shadcn/ui, jsfeat (local), Vite 8, Tailwind CSS 4

```typescript
import { readdirSync, writeFileSync, statSync } from 'fs';
import { join, relative } from 'path';

const testImagesDir = join(__dirname, '..', 'demo', 'public', 'test-images');
const outputPath = join(__dirname, '..', 'demo', 'src', 'lib', 'test-manifest.ts');

interface TestImage {
  path: string;      // URL path relative to public/
  filename: string;
  category: string;
}

const images: TestImage[] = [];

const categories = readdirSync(testImagesDir).filter((name) => {
  return statSync(join(testImagesDir, name)).isDirectory();
});

for (const category of categories.sort()) {
  const dir = join(testImagesDir, category);
  const files = readdirSync(dir)
    .filter((f) => /\.(jpe?g|png|webp)$/i.test(f))
    .sort((a, b) => {
      // Natural sort: Photo-1 before Photo-10
      const numA = parseInt(a.match(/\d+/)?.[0] ?? '0');
      const numB = parseInt(b.match(/\d+/)?.[0] ?? '0');
      return numA - numB;
    });

  for (const file of files) {
    images.push({
      path: `/test-images/${category}/${file}`,
      filename: file,
      category,
    });
  }
}

const output = `// Auto-generated by scripts/gen-test-manifest.ts — do not edit manually
// Run: npx tsx scripts/gen-test-manifest.ts

export interface TestImage {
  path: string;
  filename: string;
  category: string;
}

export const testImages: TestImage[] = ${JSON.stringify(images, null, 2)};

export const testCategories = [...new Set(testImages.map((i) => i.category))];
`;

writeFileSync(outputPath, output, 'utf-8');
console.log(\`Generated \${images.length} test images in \${categories.length} categories → \${relative(process.cwd(), outputPath)}\`);
```

- [ ] **Step 1: Create the DebugCanvas component**

Create `demo/src/components/dev/DebugCanvas.tsx`:

```tsx
import { useRef, useEffect, useCallback, useState } from 'react';
import { cardDetectionDemo, getCardDebugBuffers } from '@/lib/demos';
import { useProfiler } from '@/hooks/useProfiler';
import { Switch } from '@/components/ui/switch';
import { Label } from '@/components/ui/label';
import { Button } from '@/components/ui/button';

export interface OverlayToggles {
  canny: boolean;
  morph: boolean;
  contours: boolean;
  quad: boolean;
}

interface DebugCanvasProps {
  imageSrc: string | null;          // URL of static test image, or null for webcam
  videoRef: React.RefObject<HTMLVideoElement | null>;
  isWebcam: boolean;
  frozen: boolean;
  params: Record<string, any>;
  onMetricsUpdate: (metrics: DetectionMetrics) => void;
}

export interface DetectionMetrics {
  detected: boolean;
  debugInfo: string;
  rectFill: number;
  aspect: number;
  meanBrightness: number;
  morphThreshold: number;
  qualityScore: number;
}

export default function DebugCanvas({
  imageSrc,
  videoRef,
  isWebcam,
  frozen,
  params,
  onMetricsUpdate,
}: DebugCanvasProps) {
  const canvasRef = useRef<HTMLCanvasElement>(null);
  const overlayCanvasRef = useRef<HTMLCanvasElement>(null);
  const imageRef = useRef<HTMLImageElement | null>(null);
  const profiler = useProfiler();
  const [overlays, setOverlays] = useState<OverlayToggles>({
    canny: true,
    morph: false,
    contours: false,
    quad: true,
  });
  const [canvasSize, setCanvasSize] = useState({ w: 640, h: 480 });
  const animRef = useRef<number>(0);
  const setupDone = useRef(false);

  // Load static image
  useEffect(() => {
    if (!imageSrc || isWebcam) return;
    const img = new Image();
    img.onload = () => {
      imageRef.current = img;
      setCanvasSize({ w: img.naturalWidth, h: img.naturalHeight });
    };
    img.src = imageSrc;
  }, [imageSrc, isWebcam]);

  // Setup detection demo when canvas size changes
  useEffect(() => {
    const canvas = canvasRef.current;
    if (!canvas) return;
    canvas.width = canvasSize.w;
    canvas.height = canvasSize.h;
    const overlay = overlayCanvasRef.current;
    if (overlay) {
      overlay.width = canvasSize.w;
      overlay.height = canvasSize.h;
    }
    // Re-setup the demo with current params
    const video = videoRef.current ?? document.createElement('video');
    cardDetectionDemo.setup(canvas, video, params);
    setupDone.current = true;
    return () => {
      cardDetectionDemo.cleanup();
      setupDone.current = false;
    };
  }, [canvasSize, videoRef]);

  // Notify demo of param changes
  useEffect(() => {
    if (!setupDone.current) return;
    for (const [key, value] of Object.entries(params)) {
      cardDetectionDemo.onParamChange?.(key, value);
    }
  }, [params]);

  const runDetection = useCallback(() => {
    const canvas = canvasRef.current;
    if (!canvas || !setupDone.current) return;
    const ctx = canvas.getContext('2d', { willReadFrequently: true });
    if (!ctx) return;

    const { w, h } = canvasSize;

    // Draw source
    if (isWebcam) {
      const video = videoRef.current;
      if (!video || video.readyState < 2) return;
      ctx.drawImage(video, 0, 0, w, h);
    } else if (imageRef.current) {
      ctx.drawImage(imageRef.current, 0, 0, w, h);
    } else {
      return;
    }

    // Run the EXACT same pipeline
    const dummyVideo = (isWebcam ? videoRef.current : document.createElement('video')) as HTMLVideoElement;
    profiler.frameStart();
    cardDetectionDemo.process(ctx, dummyVideo, w, h, profiler);
    profiler.frameEnd();

    // Read debug buffers and emit metrics
    const bufs = getCardDebugBuffers();
    onMetricsUpdate({
      detected: bufs.debugInfo !== '',
      debugInfo: bufs.debugInfo,
      rectFill: bufs.lastRectFill,
      aspect: bufs.lastAspect,
      meanBrightness: 0, // computed in pipeline
      morphThreshold: bufs.prevThreshold,
      qualityScore: 0, // will be computed from canvas
    });

    // Draw overlays on the overlay canvas
    renderOverlays(w, h);
  }, [canvasSize, isWebcam, videoRef, profiler, onMetricsUpdate, overlays]);

  function renderOverlays(w: number, h: number) {
    const overlay = overlayCanvasRef.current;
    if (!overlay) return;
    const octx = overlay.getContext('2d');
    if (!octx) return;
    octx.clearRect(0, 0, w, h);

    const bufs = getCardDebugBuffers();

    // Canny edges overlay (red)
    if (overlays.canny && bufs.gray) {
      const d = bufs.gray.data;
      const imgData = octx.createImageData(w, h);
      const px = imgData.data;
      for (let i = 0; i < w * h; i++) {
        if (d[i] > 0) {
          px[i * 4] = 255;     // R
          px[i * 4 + 1] = 50;  // G
          px[i * 4 + 2] = 50;  // B
          px[i * 4 + 3] = 180; // A
        }
      }
      octx.putImageData(imgData, 0, 0);
    }

    // Morph blob overlay (yellow)
    if (overlays.morph && bufs.edges) {
      const d = bufs.edges.data;
      const imgData = octx.getImageData(0, 0, w, h);
      const px = imgData.data;
      for (let i = 0; i < w * h; i++) {
        if (d[i] > 0) {
          px[i * 4] = Math.min(255, px[i * 4] + 200);
          px[i * 4 + 1] = Math.min(255, px[i * 4 + 1] + 200);
          px[i * 4 + 2] = 0;
          px[i * 4 + 3] = Math.max(px[i * 4 + 3], 150);
        }
      }
      octx.putImageData(imgData, 0, 0);
    }
  }

  // Animation loop for webcam mode
  useEffect(() => {
    if (!isWebcam || frozen) return;
    let running = true;
    function loop() {
      if (!running) return;
      runDetection();
      animRef.current = requestAnimationFrame(loop);
    }
    animRef.current = requestAnimationFrame(loop);
    return () => {
      running = false;
      cancelAnimationFrame(animRef.current);
    };
  }, [isWebcam, frozen, runDetection]);

  // Run detection once for static images
  useEffect(() => {
    if (isWebcam || !imageRef.current) return;
    // Small delay to ensure canvas is sized
    const t = setTimeout(() => runDetection(), 50);
    return () => clearTimeout(t);
  }, [imageSrc, isWebcam, canvasSize, params, runDetection]);

  function toggleOverlay(key: keyof OverlayToggles) {
    setOverlays((prev) => ({ ...prev, [key]: !prev[key] }));
    // Re-run to update overlays for static images
    if (!isWebcam) setTimeout(() => runDetection(), 10);
  }

  return (
    <div>
      {/* Overlay toggles */}
      <div className="flex gap-4 mb-2 flex-wrap">
        {(['canny', 'morph', 'contours', 'quad'] as const).map((key) => (
          <div key={key} className="flex items-center gap-1.5">
            <Switch
              id={`overlay-${key}`}
              checked={overlays[key]}
              onCheckedChange={() => toggleOverlay(key)}
            />
            <Label htmlFor={`overlay-${key}`} className="text-xs capitalize">
              {key === 'canny' ? 'Canny (red)' : key === 'morph' ? 'Morph (yellow)' : key === 'contours' ? 'Contours (cyan)' : 'Detection (green)'}
            </Label>
          </div>
        ))}
      </div>

      {/* Stacked canvases */}
      <div className="relative border border-border rounded overflow-hidden" style={{ aspectRatio: `${canvasSize.w}/${canvasSize.h}` }}>
        <canvas
          ref={canvasRef}
          className="absolute inset-0 w-full h-full"
          style={{ imageRendering: 'auto' }}
        />
        <canvas
          ref={overlayCanvasRef}
          className="absolute inset-0 w-full h-full pointer-events-none"
          style={{ imageRendering: 'auto' }}
        />
      </div>

      {/* Profiler display */}
      <div className="mt-1 text-xs text-muted-foreground font-mono">
        {profiler.display.fps} FPS | {profiler.display.totalMs.toFixed(1)}ms |{' '}
        {profiler.display.stages.map((s) => `${s.name}: ${s.ms.toFixed(1)}`).join(' | ')}
      </div>
    </div>
  );
}
```

Create `demo/src/components/dev/PipelineStages.tsx`:

```tsx
import { useRef, useEffect } from 'react';
import { getCardDebugBuffers } from '@/lib/demos';

const STAGES = [
  { id: 'gray', label: 'Grayscale', color: false },
  { id: 'blurred', label: 'Blurred', color: false },
  { id: 'canny', label: 'Canny', color: false },
  { id: 'edges', label: 'Morph', color: true },
] as const;

interface PipelineStagesProps {
  width: number;
  height: number;
  renderTick: number; // increment to trigger re-render
}

export default function PipelineStages({ width, height, renderTick }: PipelineStagesProps) {
  const canvasRefs = useRef<(HTMLCanvasElement | null)[]>([]);

  useEffect(() => {
    const bufs = getCardDebugBuffers();
    if (!bufs.gray) return;

    const thumbW = 160;
    const thumbH = Math.round(thumbW * height / width) || 90;

    for (let si = 0; si < STAGES.length; si++) {
      const canvas = canvasRefs.current[si];
      if (!canvas) continue;
      canvas.width = thumbW;
      canvas.height = thumbH;
      const ctx = canvas.getContext('2d');
      if (!ctx) continue;

      const stage = STAGES[si];
      let srcData: Uint8Array | null = null;

      if (stage.id === 'gray') srcData = bufs.gray?.data ?? null;
      else if (stage.id === 'blurred') srcData = bufs.blurred?.data ?? null;
      else if (stage.id === 'canny') srcData = bufs.gray?.data ?? null; // canny edges saved here
      else if (stage.id === 'edges') srcData = bufs.edges?.data ?? null;

      if (!srcData) continue;

      const imgData = ctx.createImageData(thumbW, thumbH);
      const px = imgData.data;
      const scaleX = width / thumbW;
      const scaleY = height / thumbH;

      for (let y = 0; y < thumbH; y++) {
        for (let x = 0; x < thumbW; x++) {
          const srcIdx = (Math.floor(y * scaleY) * width + Math.floor(x * scaleX));
          const dstIdx = (y * thumbW + x) * 4;
          const v = srcData[srcIdx] ?? 0;

          if (stage.color) {
            // Green for morph blob
            px[dstIdx] = 0;
            px[dstIdx + 1] = v ? 180 : 20;
            px[dstIdx + 2] = 0;
          } else {
            px[dstIdx] = v;
            px[dstIdx + 1] = v;
            px[dstIdx + 2] = v;
          }
          px[dstIdx + 3] = 255;
        }
      }
      ctx.putImageData(imgData, 0, 0);
    }
  }, [renderTick, width, height]);

  return (
    <div className="space-y-2">
      <h3 className="text-xs font-semibold text-muted-foreground uppercase tracking-wider">Pipeline Stages</h3>
      {STAGES.map((stage, i) => (
        <div key={stage.id}>
          <div className="text-[10px] text-muted-foreground mb-0.5">{stage.label}</div>
          <canvas
            ref={(el) => { canvasRefs.current[i] = el; }}
            className="w-full border border-border/50 rounded"
            style={{ imageRendering: 'pixelated' }}
          />
        </div>
      ))}
    </div>
  );
}
```

import { testImages, testCategories, type TestImage } from '@/lib/test-manifest';
import { ScrollArea, ScrollBar } from '@/components/ui/scroll-area';
import { Button } from '@/components/ui/button';

### Task 8: Wire up DevPage with all components

- [ ] **Step 1: Replace the placeholder with the full implementation**

Replace the entire content of `demo/src/pages/DevPage.tsx`:

```tsx
import { useState, useCallback, useRef } from 'react';
import { useWebcam } from '@/hooks/useWebcam';
import { Button } from '@/components/ui/button';
import { Switch } from '@/components/ui/switch';
import { Label } from '@/components/ui/label';
import DebugCanvas, { type DetectionMetrics } from '@/components/dev/DebugCanvas';
import PipelineStages from '@/components/dev/PipelineStages';
import DetectionPanel, { DEFAULT_PARAMS } from '@/components/dev/DetectionPanel';
import TestImageStrip, { type Verdict } from '@/components/dev/TestImageStrip';
import { testImages } from '@/lib/test-manifest';
import { cardDetectionDemo } from '@/lib/demos';

const VERDICTS_KEY = 'detection-debug-verdicts';
const NOTES_KEY = 'detection-debug-notes';

function loadVerdicts(): Record<string, Verdict> {
  try { return JSON.parse(localStorage.getItem(VERDICTS_KEY) ?? '{}'); } catch { return {}; }
}
function saveVerdicts(v: Record<string, Verdict>) {
  localStorage.setItem(VERDICTS_KEY, JSON.stringify(v));
}
function loadNotes(): Record<string, string> {
  try { return JSON.parse(localStorage.getItem(NOTES_KEY) ?? '{}'); } catch { return {}; }
}
function saveNotes(n: Record<string, string>) {
  localStorage.setItem(NOTES_KEY, JSON.stringify(n));
}

export default function DevPage() {
  // Input mode
  const [isWebcam, setIsWebcam] = useState(false);
  const [selectedImage, setSelectedImage] = useState<string | null>(testImages[0]?.path ?? null);
  const [frozen, setFrozen] = useState(false);

  // Webcam
  const { videoRef, start, stop, isActive } = useWebcam();

  // Detection params
  const [params, setParams] = useState<Record<string, any>>({ ...DEFAULT_PARAMS });

  // Metrics
  const [metrics, setMetrics] = useState<DetectionMetrics | null>(null);
  const [renderTick, setRenderTick] = useState(0);

  // Verdicts (persisted to localStorage)
  const [verdicts, setVerdicts] = useState<Record<string, Verdict>>(loadVerdicts);
  const [allNotes, setAllNotes] = useState<Record<string, string>>(loadNotes);
  const [batchRunning, setBatchRunning] = useState(false);

  const imageSrc = isWebcam ? null : selectedImage;
  const currentVerdict: Verdict = selectedImage ? (verdicts[selectedImage] ?? 'untested') : 'untested';
  const currentNotes = selectedImage ? (allNotes[selectedImage] ?? '') : '';

  // Handlers
  function handleParamChange(key: string, value: number) {
    setParams((prev) => ({ ...prev, [key]: value }));
  }

  function handleResetParams() {
    setParams({ ...DEFAULT_PARAMS });
    // Also re-setup the demo
    cardDetectionDemo.onParamChange?.('blurKernel', DEFAULT_PARAMS.blurKernel);
    cardDetectionDemo.onParamChange?.('cannyLow', DEFAULT_PARAMS.cannyLow);
    cardDetectionDemo.onParamChange?.('cannyHigh', DEFAULT_PARAMS.cannyHigh);
    cardDetectionDemo.onParamChange?.('minContourArea', DEFAULT_PARAMS.minContourArea);
  }

  function handleMetricsUpdate(m: DetectionMetrics) {
    setMetrics(m);
    setRenderTick((t) => t + 1);
  }

  function handleVerdictChange(v: 'pass' | 'fail') {
    if (!selectedImage) return;
    const next = { ...verdicts, [selectedImage]: v };
    setVerdicts(next);
    saveVerdicts(next);
  }

  function handleNotesChange(n: string) {
    if (!selectedImage) return;
    const next = { ...allNotes, [selectedImage]: n };
    setAllNotes(next);
    saveNotes(next);
  }

  function handleToggleWebcam() {
    if (isWebcam) {
      stop();
      setIsWebcam(false);
    } else {
      setIsWebcam(true);
      start({ width: 640, height: 480 });
    }
  }

  async function handleRunAll() {
    setBatchRunning(true);
    const newVerdicts = { ...verdicts };

    for (const img of testImages) {
      setSelectedImage(img.path);
      // Wait for image to load and detection to run
      await new Promise((r) => setTimeout(r, 300));
      // Check if detection succeeded
      const m = metrics;
      newVerdicts[img.path] = m?.detected ? 'pass' : 'fail';
    }

    setVerdicts(newVerdicts);
    saveVerdicts(newVerdicts);
    setBatchRunning(false);
  }

  return (
    <div className="p-4 flex flex-col gap-4 h-[calc(100vh-48px)]">
      {/* Header */}
      <div className="flex items-center gap-4">
        <h1 className="text-lg font-bold">Detection Debug Workbench</h1>
        <div className="flex items-center gap-2">
          <Switch id="webcam-toggle" checked={isWebcam} onCheckedChange={handleToggleWebcam} />
          <Label htmlFor="webcam-toggle" className="text-xs">Webcam</Label>
        </div>
        {isWebcam && (
          <div className="flex gap-2">
            <Button size="sm" variant="outline" className="h-7 text-xs" onClick={() => setFrozen((f) => !f)}>
              {frozen ? 'Resume' : 'Freeze'}
            </Button>
          </div>
        )}
      </div>

      {/* Main content: canvas + sidebar */}
      <div className="flex gap-4 flex-1 min-h-0">
        {/* Left: Debug Canvas (60%) */}
        <div className="flex-[3] min-w-0">
          <DebugCanvas
            imageSrc={imageSrc}
            videoRef={videoRef}
            isWebcam={isWebcam}
            frozen={frozen}
            params={params}
            onMetricsUpdate={handleMetricsUpdate}
          />
        </div>

        {/* Right sidebar (40%) */}
        <div className="flex-[2] min-w-0 overflow-y-auto space-y-4">
          <PipelineStages
            width={isWebcam ? 640 : 1920}
            height={isWebcam ? 480 : 1080}
            renderTick={renderTick}
          />
          <DetectionPanel
            params={params}
            onParamChange={handleParamChange}
            onResetParams={handleResetParams}
            metrics={metrics}
            verdict={currentVerdict}
            onVerdictChange={handleVerdictChange}
            notes={currentNotes}
            onNotesChange={handleNotesChange}
          />
        </div>
      </div>

      {/* Bottom: Test Image Strip */}
      <div className="flex-shrink-0">
        <TestImageStrip
          selectedImage={selectedImage}
          onSelectImage={setSelectedImage}
          verdicts={verdicts}
          onRunAll={handleRunAll}
          running={batchRunning}
        />
      </div>
    </div>
  );
}
```
